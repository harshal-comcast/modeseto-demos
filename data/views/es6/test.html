let/var/const

classic problem

var list = [0, 1, 2, 3, 4, 5];
var parent = document.getElementById('testElement');

for (var i = 0; i < list.length; i++) {
    var elem = document.createElement('div');  
    elem.innerText = 'item ' + list[i];
	  elem.addEventListener('click', function(){ console.log(i); });
    parent.appendChild(elem);
}

Is there a cleaner way to solve this problem? YES - use let instead of var in the for loop! let gives us a fresh binding to i at each iteration, so we have a properly scoped callback. 

    (function(i) {
        var elem = document.createElement('div');  
        elem.innerText = 'item ' + list[i];
        elem.addEventListener('click', function(){ console.log(i); });
        parent.appendChild(elem);
 })(i);
